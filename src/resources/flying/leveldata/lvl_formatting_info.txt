FLYING LEVELDATA

AUTOMATIC TRIGGER:
[0] = automaticTrigger (String), [1] = hitbox.x, [2] = hitbox.y, [3] = hitbox.width
[4] = hitbox.height, [5] = cutsceneName


ALL ENTITIES:
[0] = name (String), [1] = hitbox.x, [2] = hitbox.y, [3] = direction (int), [4] = shootTimer

NAMES:
POWERUP : powerup
TARGET: target 
DRONE: drone (ShootInterval : 100 - 140)
SMALLSHIP: smallShip
OCTADRONE: octaDrone (shootTimer: 60 - 180. It only shoots once)
TANKDRONE: tankDrone
BLASTERDRONE: blasterDrone
REAPERDRONE: reaperDrone
FLAMEDRONE: flameDrone (shootTimer is set at 120. It only shoots once)
WASPDRONE: waspDrone
KAMIKAZEDRONE: kamikazeDrone
